368  ## @brief Construcs the where= ..and where=... clause
369  # @param cols [list] the columns needed to construct the clause
370  # @return returnquery [string] the And clause
371  def constructAndQuery(self,cols):
372  returnquery=""
373  if(len(cols)==0):
374  return ""
375  else:
376  for i in range(len(cols)):
377  if i==0:
378  returnquery=returnquery+cols[i].s[0]+"=\""+cols[i].s[1]+"\""

380  returnquery=returnquery+" AND "+cols[i].s[0]+"=\""+cols[i].s[1]+"\""
381  returnquery=" WHERE "+returnquery
