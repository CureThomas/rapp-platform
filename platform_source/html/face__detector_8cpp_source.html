49 std::vector<cv::Rect> FaceDetector::detectFaces(const cv::Mat& input_img)
50 {
51  std::vector<cv::Rect> front_faces, profile_faces, final_faces;
52  cv::Mat grayscale_img;
53  if( input_img.empty() )
54  {
55  return final_faces;
56  }

58  cv::equalizeHist(grayscale_img, grayscale_img);

82 std::vector<cv::Rect> FaceDetector::detectFaces(const cv::Mat& input_img,
83  const std::string& haar_path)
84 {
85  std::vector<cv::Rect> faces, final_faces;
86 
87  // The Haar cascade classifier
88  cv::CascadeClassifier face_cascade;
89  // Create the classifier
90  face_cascade.load(haar_path);
91 
92  face_cascade.detectMultiScale(input_img, faces, 1.1, 4);
93 
94  // If no faces were found make the algorithm less strict
95  if(faces.size() == 0)
96  {
97  face_cascade.detectMultiScale(input_img, faces, 1.1, 3);
98  }
99 
100  // Check the faces again to eliminate false positives
101  for(unsigned int i = 0 ; i < faces.size() ; i++)
102  {
103  cv::Rect tmp_rect = faces[i];
104  tmp_rect.x -= 10;
105  tmp_rect.y -= 10;
106  tmp_rect.width += 20;
107  tmp_rect.height += 20;
108  if(tmp_rect.x < 0 || tmp_rect.y < 0 ||
109  (tmp_rect.x + tmp_rect.width) > input_img.size().width ||
110  (tmp_rect.y + tmp_rect.height) > input_img.size().height)
111  {
112  continue;
113  }
114  cv::Mat temp_map = input_img(faces[i]);
115  std::vector<cv::Rect> tmp_faces;
116  face_cascade.detectMultiScale(temp_map, tmp_faces, 1.1, 3);
117  if(tmp_faces.size() == 0)
118  {
119  continue;
120  }
121  final_faces.push_back(faces[i]);
122  }
123  return final_faces;
124 }
