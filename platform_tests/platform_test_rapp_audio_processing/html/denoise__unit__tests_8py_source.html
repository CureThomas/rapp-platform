44  def test_realFile(self):
45  original_file = self.auxiliary_files_url + "/nai_sample.wav"
46  denoised_file = self.auxiliary_files_url + "/nai_sample_sox_denoised.wav"
47  user = 'rapp'
48  audio_type = 'nao_wav_1_ch'
49  scale = 0.2
50 
51  result = self.sox_denoise_module.soxDenoise(\
52  user,\
53  audio_type,\
54  original_file,\
55  denoised_file,\
56  scale)
57  # The function thinks the denoising succeded
58  self.assertEqual(result, "true")
59  # Check for the denoised file
60  denoised_exists = os.path.isfile(denoised_file)
61  self.assertEqual(denoised_exists, True)
62  # Check if denoised energy is lower than the initial one
63  samp_freq, signal_orig = wavfile.read(original_file)
64  energy_orig = 0.0
65  for i in signal_orig:
66  energy_orig += i * 1.0 * i
67  samp_freq, signal_denoised = wavfile.read(denoised_file)
68  energy_denoised = 0.0
69  for i in signal_denoised:
70  energy_denoised += i * 1.0 * i
71  self.assertGreater(energy_orig, energy_denoised)
72 
73  # erase the denoised file
74  os.remove(denoised_file)
